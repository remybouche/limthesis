\makeatletter
\renewcommand\chapter{%
  \if@openright
    \cleardoublepage
  \else
   \clearpage
  \fi
  \if@twocolumn
    \onecolumn
    \@tempswatrue
  \else
    \@tempswafalse
  \fi
  \secdef\@chapter\@schapter}

\def\@chapter[#1]#2{%
    \ifnum \c@secnumdepth >-2\relax
      \refstepcounter{chapter}%
      \addcontentsline{toc}{chapter}{Chapter \thechapter : \texorpdfstring{\parbox[t]{0.7\textwidth}{#1\\}}{#1}}%
    \else
      \addcontentsline{toc}{chapter}{\texorpdfstring{\parbox[t]{0.7\textwidth}{#1\\}}{#1}}%
    \fi
    \markboth{#1}{}%
    \thispagestyle{watermark}
    {\centering
     \interlinepenalty \@M
     \normalfont
     \ifnum \c@secnumdepth >-2\relax
       \null\vfil
       \huge\bfseries Chapter \thechapter
       \vskip 20\p@
     \fi
     \huge \bfseries \texorpdfstring{\parbox[t][][c]{0.7\textwidth}{ \centering #1}}{#1}\par
     \vfil\newpage}%
    %  \vfil}%
    \@endchapter}

\def\@schapter#1{%
    \thispagestyle{watermark}%
    \markboth{#1}{}%
    \phantomsection
    \addcontentsline{toc}{chapter}{#1}
    {\centering
     \interlinepenalty \@M
     \normalfont
     {\huge \bfseries {\parbox[t][][c]{0.7\textwidth}{\centering #1\\}}\par}}
     \vskip 20\p@%
    }
    
\def\@endchapter{
        \if@twoside
        \if@openright
        \null
        \newpage
         \fi
         \fi
         \if@tempswa
        \twocolumn
        \fi}
\makeatother
